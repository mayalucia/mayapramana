* Build It

The lesson defines $\rho = \frac{1}{2}(\mathbf{I} + \mathbf{M}\cdot\boldsymbol{\sigma})$.
Let us build the machinery to go from density matrix to Bloch vector
and back, and verify that the geometry holds.

** Python: Density matrix to Bloch vector

#+begin_src python
import numpy as np

# Pauli matrices
sigma_x = np.array([[0, 1], [1, 0]], dtype=complex)
sigma_y = np.array([[0, -1j], [1j, 0]], dtype=complex)
sigma_z = np.array([[1, 0], [0, -1]], dtype=complex)
I2 = np.eye(2, dtype=complex)

def bloch_vector(rho):
    """Extract the Bloch vector M from a 2x2 density matrix."""
    mx = np.trace(rho @ sigma_x).real
    my = np.trace(rho @ sigma_y).real
    mz = np.trace(rho @ sigma_z).real
    return np.array([mx, my, mz])

def density_matrix(M):
    """Construct the density matrix from a Bloch vector."""
    mx, my, mz = M
    return 0.5 * (I2 + mx * sigma_x + my * sigma_y + mz * sigma_z)

# Test: spin-up state
rho_up = np.array([[1, 0], [0, 0]], dtype=complex)
M_up = bloch_vector(rho_up)
print(f"Spin-up:   M = ({M_up[0]:.1f}, {M_up[1]:.1f}, {M_up[2]:.1f})")
print(f"  |M| = {np.linalg.norm(M_up):.1f} (pure state)")

# Test: spin-down state
rho_down = np.array([[0, 0], [0, 1]], dtype=complex)
M_down = bloch_vector(rho_down)
print(f"Spin-down: M = ({M_down[0]:.1f}, {M_down[1]:.1f}, {M_down[2]:.1f})")

# Test: spin along +x
rho_x = np.array([[0.5, 0.5], [0.5, 0.5]], dtype=complex)
M_x = bloch_vector(rho_x)
print(f"Spin +x:   M = ({M_x[0]:.1f}, {M_x[1]:.1f}, {M_x[2]:.1f})")

# Test: maximally mixed
rho_mixed = 0.5 * I2
M_mixed = bloch_vector(rho_mixed)
print(f"Mixed:     M = ({M_mixed[0]:.1f}, {M_mixed[1]:.1f}, {M_mixed[2]:.1f})")
print(f"  |M| = {np.linalg.norm(M_mixed):.4f} (maximally mixed)")
#+end_src

** Verification: round-trip

The map $\rho \to \mathbf{M} \to \rho$ must be the identity.

#+begin_src python
# Round-trip test: random pure state
theta, phi = np.pi / 3, np.pi / 4  # arbitrary angles
psi = np.array([np.cos(theta/2), np.exp(1j*phi)*np.sin(theta/2)])
rho_pure = np.outer(psi, psi.conj())

M = bloch_vector(rho_pure)
rho_reconstructed = density_matrix(M)

error = np.linalg.norm(rho_pure - rho_reconstructed)
print(f"Round-trip error: {error:.2e}")
assert error < 1e-14, "Round-trip failed"

# Check the Bloch vector matches spherical coordinates
M_expected = np.array([np.sin(theta)*np.cos(phi),
                       np.sin(theta)*np.sin(phi),
                       np.cos(theta)])
angle_error = np.linalg.norm(M - M_expected)
print(f"Bloch vector vs spherical coords: error = {angle_error:.2e}")
assert angle_error < 1e-14, "Spherical coordinate map failed"

print(f"|M| = {np.linalg.norm(M):.6f} (should be 1.0 for pure state)")
assert abs(np.linalg.norm(M) - 1.0) < 1e-14, "Pure state not on sphere surface"

print("\nAll round-trip assertions passed.")
#+end_src

** Verification: purity equals radius squared

The purity $\mathrm{Tr}(\rho^2) = \frac{1}{2}(1 + |\mathbf{M}|^2)$.
This is a physics claim: we can test it.

#+begin_src python
# Test purity relation for random mixed states
np.random.seed(42)
print(f"{'|M|':>8} {'Tr(rho^2)':>12} {'(1+|M|^2)/2':>14} {'error':>10}")

for _ in range(5):
    # Random Bloch vector inside the sphere
    M_rand = np.random.randn(3)
    M_rand *= np.random.uniform(0, 1) / np.linalg.norm(M_rand)
    rho_rand = density_matrix(M_rand)

    purity = np.trace(rho_rand @ rho_rand).real
    expected_purity = 0.5 * (1 + np.dot(M_rand, M_rand))
    error = abs(purity - expected_purity)
    print(f"{np.linalg.norm(M_rand):8.4f} {purity:12.6f} {expected_purity:14.6f} {error:10.2e}")

    assert error < 1e-14, f"Purity relation failed: {error}"

print("\nPurity relation verified for all test cases.")
#+end_src

** Verification: precession preserves |M|

The cross product $\gamma\mathbf{M}\times\mathbf{B}$ is perpendicular
to $\mathbf{M}$, so it cannot change $|\mathbf{M}|$. The lesson's
=bloch_derivative= with infinite $T_1$, $T_2$ should conserve norm.
We already tested this in the main lesson — but here we verify it at
the density matrix level.

#+begin_src python
from scipy.linalg import expm

def unitary_evolution(rho, H, dt):
    """Evolve rho under Hamiltonian H for time dt. Pure unitary."""
    U = expm(-1j * H * dt)
    return U @ rho @ U.conj().T

# Hamiltonian for Larmor precession about z
gamma = 2 * np.pi * 7e9 * 0.5  # Rb-87
Bz = 50e-6  # 50 uT
H = -0.5 * gamma * Bz * sigma_z  # H = -gamma B . S, S = sigma/2

# Start with pure state on equator
rho0 = density_matrix(np.array([1.0, 0.0, 0.0]))
M0 = bloch_vector(rho0)

# Evolve for one Larmor period
omega_L = gamma * Bz
period = 2 * np.pi / omega_L
dt = period / 100

rho_t = rho0
norms = [np.linalg.norm(bloch_vector(rho0))]

for i in range(100):
    rho_t = unitary_evolution(rho_t, H, dt)
    norms.append(np.linalg.norm(bloch_vector(rho_t)))

M_final = bloch_vector(rho_t)
print(f"Initial |M| = {norms[0]:.6f}")
print(f"Final   |M| = {norms[-1]:.6f}")
print(f"Max deviation from 1.0: {max(abs(n - 1.0) for n in norms):.2e}")
print(f"Final M: ({M_final[0]:.6f}, {M_final[1]:.6f}, {M_final[2]:.6f})")
print(f"  (should return to (1, 0, 0) after one period)")
print(f"Return error: {np.linalg.norm(M_final - M0):.2e}")

assert max(abs(n - 1.0) for n in norms) < 1e-10, "Norm not conserved"
assert np.linalg.norm(M_final - M0) < 1e-6, "Did not return after one period"
print("\nUnitary evolution preserves |M| and returns after one period.")
#+end_src
